// Математические функции
int add(int a, int b) {
    return a + b;
}

int sub(int a, int b) {
    return a - b;
}

int mul(int a, int b) {
    return a * b;
}

int div(int a, int b) {
    return a / b;
}

int mod(int a, int b) {
    return a % b;
}

// Составные операции
int calculate(int a, int b, int c) {
    return add(a, b) * sub(c, a) + mul(b, c) - div(a, b);
}

int complexCalc(int x, int y, int z) {
    return sub(mul(add(x, y), z), div(y, 2));
}

// Функция без возврата значения
void printResult(int val) {
    // Просто вызов другой функции
    printValue(val);
    return;
}

void printValue(int v) {
    print(v);
    return;
}

// Основная программа
int program() {
    // Простые вычисления
    print(add(5, 3));
    print(sub(10, 4));
    print(mul(6, 7));
    print(div(15, 3));
    print(mod(17, 5));
    
    // Сложные выражения
    print(calculate(1, 2, 3));
    print(complexCalc(4, 5, 6));
    
    // Цепочка вызовов
    print(mul(add(2, 3), sub(10, 4)));
    print(div(complexCalc(1, 2, 3), add(5, 5)));
    
    // Вызов функций без возврата значения
    printResult(42);
    printValue(100);
    
    // Возврат результата
    return add(calculate(1, 2, 3), complexCalc(4, 5, 6));
}

print(program());

// Функция без возвращаемого значения
void printMessage(int a) {
    print(a);
    return;
}

// Функция с одним параметром
int square(int n) {
    return n * n;
}

// Функция, вызывающая другие функции
int testFunctions(int val) {
    square(val);
    max(1, 2);
    return val;
}

// Точка входа
void main() {
    printMessage(-100);
    square(7);
    print(testFunctions(42));
    return;
}

main();

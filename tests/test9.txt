// Функция для демонстрации
int doubleValue(int n) {
    print(n);
    return n * 2;
}

int square(int n) {
    print(n);
    return n * n;
}

int main(int i) {
    
    // Простой вывод
    print(123);
    
    // Вывод выражений
    print(10 + 20);
    print(50 - 15);
    print(6 * 7);
    print(100 / 4);
    print(17 % 5);
    
    // Вызов функций с print внутри
    doubleValue(5);
    square(4);
    
    // Вложенные вызовы с print
    print(doubleValue(3));
    print(square(doubleValue(2)));
    
    // Сложные выражения
    print(doubleValue(5) + square(3) * 2);
    print((doubleValue(2) + square(3)) / 2);
    
    // Массовый вывод
    print(i);
    print(i + 1);
    print(i + 2);
    print(i + 3);
    print(i + 4);
    
    return 0;
}

main(1);

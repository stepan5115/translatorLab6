// Функция без параметров и без тела
void empty() {}

// Различные типы
int getValue() {
    return -42;
}

void process(int a, int b, int c) {
    // Пустое тело
}

int compute(int x, int y) {
    // Сложное выражение со всеми операторами
    return (x + y) * (x - y) / 2 + 100 % 3;
}

int nested(int n) {
    // Вложенные вызовы и сложные выражения
    return compute(n + 5, n * 2) * square(n);
}

int square(int n) {
    return n * n;
}

// Основная программа
int main() {
    // Объявления функций
    empty();
    
    // Вызовы с разным количеством аргументов
    print(abs(getValue()));
    process(1, 2, 3);
    
    // Вычисления
    print(compute(10, 5));
    print(nested(7));
    
    // Цепочка вызовов
    print(fact(square(compute(3, 4))));
    
    return 0;
}

main();
